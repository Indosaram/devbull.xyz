def solution(lottos, win_nums):
    unknown = 0
    win = []
    lose = []
    for num in lottos:
        if num == 0:
            unknown += 1
        elif num in win_nums:
            win.append(num)
            win_nums.remove(num)
        else:
            lose.append(num)

    max_win = len(win) + unknown
    min_win = len(win)

    stand = {6: 1, 5: 2, 4: 3, 3: 4, 2: 5, 1: 6, 0: 6}
    return [stand[max_win], stand[min_win]]
